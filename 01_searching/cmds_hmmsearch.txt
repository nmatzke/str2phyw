# Extracting proteins from hits table to a FASTA file of the hits

# Make a fast lookup table with Easel (esl)
cd ~/GitHub/str2phyw/01_searching/genomes/
esl-sfetch --index 34_genomes_AAs.fasta

# Copy to main directory
cp 34_genomes_AAs.fasta ~/GitHub/str2phyw/01_searching/
cp 34_genomes_AAs.fasta.ssi ~/GitHub/str2phyw/01_searching/

cd ~/GitHub/str2phyw/01_searching/
grep -v "^#" MotA_ExbB_TolQ_fromAll_seqHitsTable.txt | awk '{print $1}' | esl-sfetch -f ~/GitHub/str2phyw/01_searching/genomes/34_genomes_AAs.fasta - > AQBs_wLitLinks_aln.fasta

head  AQBs_wLitLinks_aln.fasta
wc AQBs_wLitLinks_aln.fasta
