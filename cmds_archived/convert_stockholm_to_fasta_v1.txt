# Convert to aligned clustal format, then to FASTA
esl-reformat --informat stockholm clustal FliMN_C_hmmeralign.sto | tee FliMN_C_hmmeralign.sto.clustal
clustalw2fasta FliMN_C_hmmeralign.sto.clustal > FliMN_C_hmmeralign.sto.fasta
