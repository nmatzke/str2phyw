# NOTE ON PERMISSIONS ON A MAC
#######################################################
On Macs, newly downloaded software will often give messages like this:

=========
Apple could not verify “famsa2” is free of malware that may harm your Mac or compromise your privacy.
=========

You need to give your computer permission to run this file.  Go to:

System Settings -> Privacy & Security -> scroll to bottom -> "'famsa2' was blocked to protect your Mac." -> Allow Anyway (put in password if needed)

# Install rBLAST
# https://github.com/mhahsler/rBLAST
# 1. Get BLAST tools from:
#    https://blast.ncbi.nlm.nih.gov/Blast.cgi?PAGE_TYPE=BlastDocs&DOC_TYPE=Download
#
# 2. Install .dmg to: 
#    cd /usr/local/ncbi/blast/bin
#
# blast_formatter		deltablast		rpsblast
# blastdb_aliastool	dustmasker		rpstblastn
# blastdbcheck		get_species_taxids.sh	segmasker
# blastdbcmd		legacy_blast.pl		tblastn
# blastn			makeblastdb		tblastx
# blastp			makembindex		update_blastdb.pl
# blastx			makeprofiledb		windowmasker
# convert2blastmask	psiblast
#
# 3. Install rBLAST:
library(devtools)
install_github("mhahsler/rBLAST") 
library(rBLAST)
